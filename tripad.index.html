        // 診断開始

        function startQuiz() {

            const daysInput = document.getElementById('input-days').value;

            const budgetInput = document.getElementById('input-budget').value;


            if(!daysInput || !budgetInput) {

                alert("日数と予算を入力してください");

                return;

            }


            userDays = daysInput;

            userBudget = budgetInput;


            // 画面切り替え

            document.getElementById('screen-intro').classList.add('hidden');

            document.getElementById('screen-quiz').classList.remove('hidden');


            loadQuestion();

        }


        // 質問の読み込み

        function loadQuestion() {

            const q = questions[currentQuestionIndex];

            document.getElementById('question-text').innerText = q.text;

            

            const container = document.getElementById('options-container');

            container.innerHTML = ""; // クリア


            q.options.forEach(opt => {

                const btn = document.createElement('button');

                btn.className = 'btn btn-option';

                btn.innerText = opt.text;

                btn.onclick = () => selectOption(opt.type);

                container.appendChild(btn);

            });


            // プログレスバー更新

            const progress = ((currentQuestionIndex) / questions.length) * 100;

            document.getElementById('progress-bar').style.width = progress + "%";

        }


        // 選択肢クリック時の処理

        function selectOption(type) {

            scores[type]++; // スコア加算

            

            currentQuestionIndex++;


            if (currentQuestionIndex < questions.length) {

                loadQuestion();

            } else {

                showResult();

            }

        }


        // 結果表示

        function showResult() {

            document.getElementById('screen-quiz').classList.add('hidden');

            document.getElementById('screen-result').classList.remove('hidden');


            // 最高得点のタイプを判定

            let maxType = "type1";

            let maxScore = 0;

            

            for (const [key, value] of Object.entries(scores)) {

                if (value > maxScore) {

                    maxScore = value;

                    maxType = key;

                }

            }


            const resultData = results[maxType];


            // テキストの埋め込み

            document.getElementById('result-title').innerText = resultData.title;

            document.getElementById('result-desc').innerText = resultData.desc;

            

            // おすすめプランの生成（入力データと組み合わせる）

            const planText = `今回は予算${userBudget}、${userDays}とのことですね。\n` +

                             `そんなあなたには、${resultData.places}がおすすめです！`;

            

            document.getElementById('result-plan').innerText = planText;

            document.getElementById('display-days').innerText = userDays;

            document.getElementById('display-budget').innerText = userBudget;

        }
